\begin{itemize}
\item{Water and roads}
\pause 
\begin{itemize}
\item[$\rightarrow$] Slows down firespread significantly or completely stops the spread
\end{itemize}
\pause 
\item{Extended Neighborhoods}
\pause 
\begin{itemize}
\item[$\rightarrow$]  Speeds up firespread
\end{itemize}
\pause
\item{Firefighting}
\pause 
\begin{itemize}
\item[$\rightarrow$]  Doesnt slow down fire
\end{itemize}
\pause 
\item{Grids}
\begin{itemize}
\item[$\rightarrow$]  Triangle grid is best for small neighborhood
\item[$\rightarrow$]  Cartesian grid is best for extended neighborhood
\end{itemize}
\end{itemize}
\end{frame}
